One-time setup

Google Sheets integration

Connect this form to your Google Sheet in about 5 minutes.

Step 1 — Create your Google Sheet
Open Google Sheets and create a new spreadsheet. Name it Cinco Patricio's Sales Log.
Add this header row in row 1 (one value per cell, A through L): Timestamp | Week Ending | Rep | Venue | Venue Type | Calls | Contacts | Sales | Wholesaler | Notes | Submitted At | Row ID
Step 2 — Create a Google Apps Script Web App
In your Sheet, click Extensions → Apps Script. Delete any existing code and paste this. This version includes a doGet function so the dashboard can read the live bottle total directly from your Sheet on every load:
function doPost(e) { var sheet = SpreadsheetApp .getActiveSpreadsheet() .getSheetByName('Sheet1'); var data = JSON.parse(e.postData.contents); sheet.appendRow([ new Date(), data.weekEnding, data.rep, data.venue, data.venueType, data.calls, data.contacts, data.sales, data.wholesaler, data.notes, data.submittedAt, Utilities.getUuid() ]); return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { var sheet = SpreadsheetApp .getActiveSpreadsheet() .getSheetByName('Sheet1'); var rows = sheet.getDataRange().getValues(); // Column H (index 7) is Sales. Row 0 is the header row. var baseline = 669; // Jan 200 + Feb 120 + Mar 349 var totalSold = baseline; var monthSales = 0; var currentMonth = new Date().getMonth(); // 0=Jan for (var i = 1; i < rows.length; i++) { var val = parseInt(rows[i][7]); if (!isNaN(val)) { totalSold += val; // Also tally current month from col B (weekEnding date string) var rowDate = new Date(rows[i][1]); if (!isNaN(rowDate) && rowDate.getMonth() === currentMonth) { monthSales += val; } } } var result = { status: 'ok', totalSold: totalSold, monthSales: monthSales }; return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); }
Step 3 — Deploy as Web App
Click Deploy → New deployment. Set:
— Type: Web App
— Execute as: Me
— Who has access: Anyone
Copy the Web App URL and paste it below, then click Save.
✓ URL saved — check-ins will now write to your Sheet.
Step 4 — Automate the Friday email (optional)
In Apps Script, add a second function to read the week's rows and send a summary email via MailApp.sendEmail(). Set a time-driven trigger for Friday at 5pm. This keeps the digest fully automated with no extra tools.
